Click the buttons to play or pause the music.

$(document).ready(function () { $('.outer').click(function () { if (change) { $(".inner").animate({ left: '70%' }); $("body").css("background-color", "black"); $("p").css("color", "white"); change = false; } else { $(".inner").animate({ left: '0px' }); $("body").css("background-color", "white"); $("p").css("color", "black"); change = true; } }); });
